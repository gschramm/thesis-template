%%---- Headers & footers -------------------------------------------------

% Wrapped in \AtBeginDocument so this runs after all package \AtBeginDocument
% hooks (thumbs, biblatex, titlesec) and cannot be overridden by them.
\AtBeginDocument{%
  % \markboth sets: leftmark  = "Chapter N"  (shown on left/even pages)
  %                 rightmark = chapter title (shown on right/odd pages)
  \renewcommand{\chaptermark}[1]{%
    \markboth{Chapter~\arabic{chapter}}{#1}%
    \addthumb{#1}{\thechapter}{white}{thesisblue}%
  }%
  \renewcommand{\sectionmark}[1]{}% keep rightmark = chapter title, not section
  \pagestyle{fancy}%
  \fancyhf{}%
  \fancyhead[LE]{\small\textcolor{headergray}{\leftmark}}%   left-aligned on left pages
  \fancyhead[RO]{\small\textcolor{headergray}{\nouppercase{\rightmark}}}% right-aligned on right pages
}
