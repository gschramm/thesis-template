\section{Algorithm pseudocode}

Pseudocode is typeset using the \texttt{algorithm} float together
with the \texttt{algpseudocode} environment from the
\texttt{algorithmicx} bundle.
Line numbers are enabled by passing \texttt{[1]} to
\texttt{algorithmic}.
\Cref{alg:sgd} implements the update rule from
\cref{eq:ch2-sgd,eq:ch2-lr}.

\begin{algorithm}[t]
  \caption{Mini-batch stochastic gradient descent}
  \label{alg:sgd}
  \begin{algorithmic}[1]
    \Require Training set $\mathcal{D}$, initial learning rate
             $\eta_{0}$, decay $\gamma$, step size $s$, batch
             size $B$
    \Ensure  Trained parameters $\boldsymbol{\theta}$
    \State Initialise $\boldsymbol{\theta}$ randomly
    \For{$t = 0, 1, 2, \ldots$}
      \State Sample mini-batch
             $\mathcal{B} \subset \mathcal{D}$,\;
             $|\mathcal{B}| = B$
      \State $\hat{g} \gets \dfrac{1}{B}
             \displaystyle\sum_{(\mathbf{x},y)\in\mathcal{B}}
             \nabla_{\!\boldsymbol{\theta}}\,
             \ell\!\left(f_{\boldsymbol{\theta}}(\mathbf{x}),
             y\right)$
             \Comment{gradient estimate}
      \State $\eta_{t} \gets \eta_{0}\cdot
             \gamma^{\lfloor t/s \rfloor}$
             \Comment{decayed rate, \cref{eq:ch2-lr}}
      \State $\boldsymbol{\theta} \gets \boldsymbol{\theta}
             - \eta_{t}\,\hat{g}$
             \Comment{parameter update, \cref{eq:ch2-sgd}}
    \EndFor
    \State \Return $\boldsymbol{\theta}$
  \end{algorithmic}
\end{algorithm}

\section{Further equations and units}

The cross-entropy loss for a $K$-class classifier is

\begin{equation}
  \mathcal{L}(\boldsymbol{\theta}) =
  -\frac{1}{N}\sum_{i=1}^{N}
    \sum_{k=1}^{K} y_{ik}\,\log\hat{p}_{ik}(\boldsymbol{\theta}),
  \label{eq:ch2-crossentropy}
\end{equation}

where $y_{ik} \in \{0,1\}$ is the ground-truth label and
$\hat{p}_{ik}$ is the predicted probability for class~$k$.
Minimising~\eqref{eq:ch2-crossentropy} via stochastic gradient
descent gives the update rule

\begin{align}
  \boldsymbol{\theta}^{(t+1)}
    &= \boldsymbol{\theta}^{(t)}
       - \eta\,\nabla_{\!\boldsymbol{\theta}}\,
         \mathcal{L}\!\left(\boldsymbol{\theta}^{(t)}\right),
       \label{eq:ch2-sgd} \\
  \eta_{t}
    &= \eta_{0}\cdot\gamma^{\lfloor t/s \rfloor},
       \label{eq:ch2-lr}
\end{align}

where $\eta_{0}$ is the initial learning rate, $\gamma < 1$ is the
decay factor, and $s$ is the step size~\cite{lecun1998}.

Use the \texttt{siunitx} package for all physical quantities and
units to ensure consistent spacing and upright unit symbols.
For example, the experiment ran for \SI{72}{\hour} on a GPU with
\SI{40}{\giga\byte} of memory, achieving an accuracy of
\SI{88.1}{\percent} after \num{50000} gradient steps.
